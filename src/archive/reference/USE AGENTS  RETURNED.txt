/**
 * Base Agent Class
 * All specialized agents extend from this
 */
export class BaseAgent {
  constructor(name, config = {}) {
    this.name = name;
    this.config = config;
    this.state = {};
    this.memory = [];
    this.maxMemory = config.maxMemory || 100;
  }

  /**
   * Process incoming messages
   */
  async process(message) {
    this.addToMemory(message);
    return await this.think(message);
  }

  /**
   * Agent's decision-making logic (override in subclasses)
   */
  async think(message) {
    throw new Error('Think method must be implemented by subclass');
  }

  /**
   * Add message to agent memory
   */
  addToMemory(message) {
    this.memory.push({
      timestamp: Date.now(),
      content: message
    });
    
    // Trim memory if too large
    if (this.memory.length > this.maxMemory) {
      this.memory.shift();
    }
  }

  /**
   * Get relevant context from memory
   */
  getContext(query, limit = 5) {
    return this.memory.slice(-limit);
  }

  /**
   * Update agent state
   */
  updateState(updates) {
    this.state = { ...this.state, ...updates };
  }

  /**
   * Get current state
   */
  getState() {
    return { ...this.state };
  }
}

    super('GameMaster', config);
    this.activeQuests = new Map();
    this.playerSessions = new Map();
    this.eventQueue = [];
  }

  async think(message) {
    const { type, playerId, data } = message;

    switch (type) {
      case 'START_GAME':
        return await this.startGame(playerId, data);
      
      case 'PLAYER_ACTION':
        return await this.handlePlayerAction(playerId, data);
      
      case 'QUEST_UPDATE':
        return await this.updateQuest(playerId, data);
      
      case 'REQUEST_HINT':
        return await this.provideHint(playerId, data);
      
      default:
        return { status: 'unknown_command', message: message };
    }
  }

  /**
   * Start a new game session
   */
  async startGame(playerId, playerData) {
    const session = {
      playerId,
      startTime: Date.now(),
      currentLevel: 1,
      score: 0,
      questsCompleted: 0,
      hintsUsed: 0,
      playerData
    };

    this.playerSessions.set(playerId, session);

    // Generate initial quest
    const initialQuest = await this.generateQuest(playerId, 1);

    return {
      status: 'game_started',
      session,
      quest: initialQuest,
      welcome: this.getWelcomeMessage(playerData)
    };
  }

  /**
   * Generate welcome message
   */
  getWelcomeMessage(playerData) {
    return {
      title: '✨ Welcome to Holy Quest',
      message: `Greetings, ${playerData.name}! Your spiritual journey begins now.`,
      instructions: [
        'Complete quests to earn points',
        'Use hints wisely - they cost points',
        'Share your wisdom with others',
        'Level up by demonstrating understanding'
      ]
    };
  }

  /**
   * Generate quest for player
   */
  async generateQuest(playerId, level) {
    const quest = {
      id: `quest_${Date.now()}`,
      playerId,
      level,
      type: this.selectQuestType(level),
      status: 'active',
      startTime: Date.now(),
      attempts: 0,
      maxAttempts: 3
    };

    this.activeQuests.set(quest.id, quest);

    return quest;
  }

  /**
   * Select quest type based on level
   */
  selectQuestType(level) {
    const types = [
      'verse_memory',
      'teaching_application',
      'wisdom_puzzle',
      'parable_interpretation',
      'moral_dilemma'
    ];

    // Weight types based on level
    const weights = {
      'verse_memory': level < 3 ? 0.4 : 0.2,
      'teaching_application': level < 5 ? 0.3 : 0.25,
      'wisdom_puzzle': level > 2 ? 0.2 : 0.1,
      'parable_interpretation': level > 3 ? 0.2 : 0.05,
      'moral_dilemma': level > 4 ? 0.2 : 0
    };

    const random = Math.random();
    let cumulative = 0;

    for (const [type, weight] of Object.entries(weights)) {
      cumulative += weight;
      if (random < cumulative) {
        return type;
      }
    }

    return types[0];
  }

  /**
   * Handle player action
   */
  async handlePlayerAction(playerId, data) {
    const session = this.playerSessions.get(playerId);
    if (!session) {
      return { status: 'error', message: 'No active session' };
    }

    const { action, questId, answer } = data;

    if (action === 'SUBMIT_ANSWER') {
      return await this.evaluateAnswer(playerId, questId, answer);
    }

    return { status: 'action_processed', data };
  }

    const quest = this.activeQuests.get(questId);
    const session = this.playerSessions.get(playerId);

    if (!quest || !session) {
      return { status: 'error', message: 'Invalid quest or session' };
    }

    quest.attempts++;

    // Here you would integrate with ContentAgent for evaluation

    if (evaluation.correct) {
      // Award points
      const points = this.calculatePoints(quest, evaluation);
      session.score += points;
      session.questsCompleted++;
      quest.status = 'completed';

      // Check for level up
      const levelUp = this.checkLevelUp(session);

      return {
        status: 'correct',
        points,
        totalScore: session.score,
        feedback: evaluation.feedback,
        levelUp,
        nextQuest: levelUp ? await this.generateQuest(playerId, session.currentLevel) : null
      };
    } else {
      const attemptsRemaining = quest.maxAttempts - quest.attempts;

      if (attemptsRemaining > 0) {
        return {
          status: 'incorrect',
          attemptsRemaining,
          feedback: evaluation.feedback,
          hint: quest.attempts === 2 ? await this.generateHint(quest) : null
        };
      } else {
        quest.status = 'failed';
        return {
          status: 'quest_failed',
          correctAnswer: evaluation.correctAnswer,
          explanation: evaluation.explanation,
          nextQuest: await this.generateQuest(playerId, session.currentLevel)
        };
      }
    }
  }

/**
 * Balance Agent
 * Monitors game difficulty and adjusts dynamically
 */
export class BalanceAgent extends BaseAgent {
  constructor(config = {}) {
    super('Balance', config);
    this.metrics = {
      averageSuccessRate: 0,
      averageAttempts: 0,
      questCompletionTimes: [],
      playerLevelDistribution: new Map()
    };
    this.targetSuccessRate = 0.65; // 65% target success rate
    this.adjustmentInterval = config.adjustmentInterval || 100; // Adjust every 100 quests
    this.questsAnalyzed = 0;
  }

  async think(message) {
    const { type, data } = message;

    switch (type) {
      case 'ANALYZE_QUEST_RESULT':
        return await this.analyzeQuestResult(data);
      
      case 'GET_DIFFICULTY_ADJUSTMENT':
        return this.getDifficultyAdjustment(data);
      
      case 'GET_BALANCE_REPORT':
        return this.getBalanceReport();
      
      default:
        return { status: 'unknown_request' };
    }
  }

  /**
   * Analyze quest result and update metrics
   */
  async analyzeQuestResult(data) {
    const { playerId, questId, success, attempts, timeSpent, level } = data;

    // Update quest counter
    this.questsAnalyzed++;

    // Update success rate
    const currentSuccessRate = this.metrics.averageSuccessRate;
    this.metrics.averageSuccessRate = 
      (currentSuccessRate * (this.questsAnalyzed - 1) + (success ? 1 : 0)) / this.questsAnalyzed;

    // Update average attempts
    const currentAvgAttempts = this.metrics.averageAttempts;
    this.metrics.averageAttempts = 
      (currentAvgAttempts * (this.questsAnalyzed - 1) + attempts) / this.questsAnalyzed;

    // Update completion times
    this.metrics.questCompletionTimes.push(timeSpent);
    if (this.metrics.questCompletionTimes.length > 1000) {
      this.metrics.questCompletionTimes.shift(); // Keep last 1000
    }

    // Update level distribution
    const levelCount = this.metrics.playerLevelDistribution.get(level) || 0;
    this.metrics.playerLevelDistribution.set(level, levelCount + 1);

    // Check if adjustment needed
    if (this.questsAnalyzed % this.adjustmentInterval === 0) {
      const adjustment = this.calculateAdjustment();
      
      return {
        status: 'analyzed',
        metrics: this.getMetricsSummary(),
        adjustment,
        shouldAdjust: adjustment.recommended
      };
    }

    return {
      status: 'analyzed',
      metrics: this.getMetricsSummary()
    };
  }

  /**
   * Calculate difficulty adjustment
   */
  calculateAdjustment() {
    const successRate = this.metrics.averageSuccessRate;
    const avgAttempts = this.metrics.averageAttempts;
    const targetRate = this.targetSuccessRate;

    const adjustment = {
      recommended: false,
      direction: null,
      magnitude: 0,
      reason: ''
    };

    // Too easy (success rate too high)
    if (successRate > targetRate + 0.15) {
      adjustment.recommended = true;
      adjustment.direction = 'increase';
      adjustment.magnitude = Math.min((successRate - targetRate) * 2, 0.3);
      adjustment.reason = `Success rate (${(successRate * 100).toFixed(1)}%) is too high. Increasing difficulty.`;
    }
    // Too hard (success rate too low)
    else if (successRate < targetRate - 0.15) {
      adjustment.recommended = true;
      adjustment.direction = 'decrease';
      adjustment.magnitude = Math.min((targetRate - successRate) * 2, 0.3);
      adjustment.reason = `Success rate (${(successRate * 100).toFixed(1)}%) is too low. Decreasing difficulty.`;
    }
    // Check average attempts
    else if (avgAttempts > 2.5) {
      adjustment.recommended = true;
      adjustment.direction = 'decrease';
      adjustment.magnitude = 0.1;
      adjustment.reason = `Average attempts (${avgAttempts.toFixed(1)}) is high. Slight difficulty decrease.`;
    } else if (avgAttempts < 1.3) {
      adjustment.recommended = true;
      adjustment.direction = 'increase';
      adjustment.magnitude = 0.1;
      adjustment.reason = `Average attempts (${avgAttempts.toFixed(1)}) is low. Slight difficulty increase.`;
    }

    return adjustment;
  }

  /**
   * Get difficulty adjustment for specific player
   */
  getDifficultyAdjustment(data) {
    const { playerId, currentLevel, recentPerformance } = data;

    // Analyze player's recent performance
    const playerSuccessRate = recentPerformance.filter(r => r.success).length / recentPerformance.length;
